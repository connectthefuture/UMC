57  static TimeCode::eStandardFrameRates ConvertToStandardFrameRate( const FrameRate & frameRate ) {
58  UMC_Uns64 intFrameRate = frameRate.Numerator() / frameRate.Denominator();
59  bool exactMatch = intFrameRate * frameRate.Denominator() == frameRate.Numerator();
60 
61  if ( exactMatch ) {
62  switch ( intFrameRate ) {
63  case 24:
64  return TimeCode::k24Fps;
65  break;
66 
67  case 25:
68  return TimeCode::k25Fps;
69  break;
70 
71  case 30:
72  return TimeCode::k30Fps;
73  break;
74 
75  case 50:
76  return TimeCode::k50Fps;
77  break;
78 
79  case 60:
80  return TimeCode::k60Fps;
81  break;
82  }
83  }
84 
85  double dblFrameRate = static_cast< double >( frameRate.Numerator() );
86  dblFrameRate /= frameRate.Denominator();
87 
88  if ( dblFrameRate > 23.975 && dblFrameRate < 23.977 ) {
89  if ( frameRate.Numerator() % 24000 == 0 && frameRate.Denominator() % 1001 == 0 &&
90  frameRate.Numerator() / 24000 == frameRate.Denominator() / 1001 )
91  return TimeCode::k23_976Fps;
92  } else if ( dblFrameRate > 29.96 && dblFrameRate < 29.98 ) {
93  if ( frameRate.Numerator() % 30000 == 0 && frameRate.Denominator() % 1001 == 0 &&
94  frameRate.Numerator() / 30000 == frameRate.Denominator() / 1001 )
95  return TimeCode::k29_97Fps;
96  } else if ( dblFrameRate > 59.93 && dblFrameRate < 59.95 ) {
97  if ( frameRate.Numerator() % 60000 == 0 && frameRate.Denominator() % 1001 == 0 &&
98  frameRate.Numerator() / 60000 == frameRate.Denominator() / 1001 )
99  return TimeCode::k59_94Fps;
100  }
101 
102  return TimeCode::kNonStandardFrameRate;
103  }
